void f5()
{
    int i, j, mas[10][10], s=0, k1=0, k2=0;
    double sr, r=0;
    for(i=0;i<10;i++){
        for(j=0;j<10;j++){
            mas[i][j]=rand()%100-50;
            s+=mas[i][j];
        }
    }
    for(i=0;i<10;i++){
        for(j=0;j<10;j++)
            cout<<mas[i][j]<<' ';
        cout<<endl;
    }
    sr=s/100.0;
    r=abs(mas[0][0]-sr);
    for(i=0;i<10;i++){
        for(j=0; j<10;j++){
            if(abs(mas[i][j]-sr)<r){
                r=abs(mas[i][j]-sr);
                k1=i;
                k2=j;
            }
        }
    }
    cout<<k1<<' '<<k2<<' '<<sr;
}
