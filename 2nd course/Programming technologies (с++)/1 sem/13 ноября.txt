{

double f2(int i)
{
    double sum=0;
	for (int g=1; g<=i; g++)
    	sum+=(g+1)/(2.0*g);
	return sum;
}

}

void f6(double &a, double &b, double &c)
{
    if (a > b) {
    double temp = a;
    a = b;
    b = temp;
  }
    if (a > c) {
    double temp = a;
    a = c;
    c = temp;
  }
    if (b > c) {
    double temp = b;
    b = c;
    c = temp;
  }
}

void f7(double *a,int n)
{
    for (int i=1; i<=n; i++)
		a[i]=(i+1)/(2.0*i);
}

void f8(int *a,int n,int &max,int &min,int &sum,int &imax,int &imin)
{
    max=min=a[0];
	imax=imin=0;
	sum=0;
	for (int i=0; i<n; i++){
		if(a[i]>max){
			max=a[i];
			imax=i;
		}
		if(a[i]<min){
			min=a[i];
			imin=i;
		}
		if(a[i]>0)
			sum+=a[i];
	}
}

double f9(int n)
{
    double fact=1;
    for (int i=2; i<=n; i++)
        fact*=i;
    return fact;
}
