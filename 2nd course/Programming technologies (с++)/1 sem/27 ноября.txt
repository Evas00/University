#include <iostream>
using namespace std;

double f1(double x, double e, int& n){
  int t=-1, j=1;
  double a, x2=x, sum=0;
  a=(t*x2)/j;
  sum+=a;
  while(abs(a)>e){
    ++n;
    x2*=x;
    j*=n;
    t*=-1;
    a=(t*x2)/(j*1.0);
    sum+=a;
  }
  return sum;
}

double f2(double* mas, int k, int& no) {
  double sum=0;
  for (int i=0; i<k;i++) {
    if(mas[i]<0){
      sum+=mas[i];
      ++no;
    }
  }
  return sum;
}

void f3(double* mas, int k) {
  for(int i=0;i<k;i++) {
    if(mas[i]<0)
        mas[i]=(3.0*i)/((2.0*i)+1.0);
  }
}

    cout<<mas[i]<<' ';

}
