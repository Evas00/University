string f1(string s1, string s2)
{
    return s1+s2;
}

string f2(string st, char sym)
{
    string dop;
    for(int i=0;i<st.length();i++){
        dop+=st[i];
        if(st[i]==sym)
            dop+=st[i];
    }
    return dop;
}

bool f3(string st, char sym, int& v1, int& v2, int& k)
{
    for(int i=0;i<st.length();i++){
        if(st[i]==sym){
            k+=1;
            if(v1==0)
                v1=i+1;
            v2=i+1;
        }
    }
    if(k!=0)
        return true;
    else
        return false;
}

string f4(string s1, string s2, int n)
{
    string dop;
    int i;
    if(n==0)
        dop=s2;
    for(i=0;i<s1.length();i++){
        dop+=s1[i];
        if(i==(n-1))
            dop+=s2;
    }
    if(n>i)
        dop+=s2;
    return dop;
}

string f5(string st, int n)
{
    string dop;
    if((n-1)<=st.length())
        for(int i=0;i<n;++i)
            dop+=st[i];
    else{
        int raz=n-st.length();
        for(int i=0;i<raz;i++)
            dop+='.';
        dop+=st;
    }
    return dop;
}

string f6(int n1, int n2, string s1,  string s2)
{
    string dop;
    for(int i=0;i<n1;++i)
        dop+=s1[i];
    for(int i=s2.length()-n2;i<s2.length();i++)
        dop+=s2[i];
    return dop;
}

bool f7(string s, string s0)
{
    int j, flag;
    for(int i=0;i<s.size()-s0.size()+1;i++){
        if (s[i]==s0[0])
        {
            j=1;
            flag=true;
            while((j<s0.size())and(flag))
                if (s[i+j]!=s0[j])
                    flag=false;
                else
                    j++;
            if (flag)
                return true;
        }
    }
    return false;
}

int f8(string s,  string s0)
{
    int j, flag, sum = 0;
    for(int i=0;i<s.size()-s0.size()+1;i++){
        if(s[i]==s0[0])
        {
            j=1;
            flag=true;
            while((j<s0.size())and(flag))
                if (s[i+j]!=s0[j])
                    flag=false;
                else
                    j++;
            if(flag)
                sum++;
        }
    }
    return sum;
}
