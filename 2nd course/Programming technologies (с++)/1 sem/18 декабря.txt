#include<iostream>
#include<math.h>
using namespace std;

double sin_moy(double z, double e){
    double sum=z+1,arg=z+1;
    int i=1;
    while (abs(arg)>e){
        i+=2;
        arg*=((-z)*z)/(i*(i-1));
        sum+=arg;
    }
    return sum;
}

double cos_moy(double z, double e){
    double sum=1,arg=1;
    int i=0;
    while (abs(arg)>e){
        i+=2;
        arg*=((-z)*z)/(i*(i-1));
        sum+=arg;
    }
    return sum;
}

double e_moy(double z, double e){
    double sum=1+z,arg=z;
    int i=1;
    while (abs(arg)>e){
        i+=1;
        arg*=z/i;
        sum+=arg;
    }
    return sum;
}

double ln_moy(double z, double e){
    double sum=z-1,arg=z-1;
    int i=1;
    while (abs(arg)>e){
        i+=1;
        arg*=(-z);
        sum+=arg/(i*1.0);
    }
    return sum;
}

int main()
{
    double z, e;
    int n=1;
    cout<<"Саргсян 18.12.2024";
    while(n){
        cin>>n;
        cin>>z>>e;
        switch(n)
        {
            case 1:{ 
                cout<<sin_moy(z,e)<<' '<<sin(z)<<endl; 
                break;
            }
            case 2:{ 
                cout<<cos_moy(z,e)<<' '<<cos(z)<<endl; 
                break;
            }
            case 3:{ 
                cout<<e_moy(z,e)<<' '<<exp(z)<<endl; 
                break;
            }
            case 4:{ 
                cout<<ln_moy(z,e)<<' '<<log10(z)<<endl; 
                break;
            }
        }
    }
    return 0;
}
